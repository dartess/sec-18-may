Подключение шрифта

@font-face{}

@font-face {
  font-family: 'Geometria';
  font-weight: 400;
  font-style: normal;
  src:
    local('Geometria'),
    local('Geometria Regular'),
    url(/local/assets/fonts/geometria/normal.woff2) format("woff2"),
    url(/local/assets/fonts/geometria/normal.woff) format("woff")
}
